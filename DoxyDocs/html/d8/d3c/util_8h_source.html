00052 template <typename T> inline T smoothInterpolate(const T &from, const T &to, T &vel, float smoothTime, float dt)
00053 {
00054         float omega = 2.f/smoothTime;
00055         float x = omega*dt;
00056         float exp = 1.f/(1.f+x+0.48f*x*x+0.235f*x*x*x);
00057         T change = from - to;
00058         T temp = (vel+omega*change)*dt;
00059         vel = (vel - omega*temp)*exp;  // Equation 5
00060         return to + (change+temp)*exp; // Equation 4
00061 }
