00047    static inline float noise(float x, float y, float z)
00048    {
00049           int X = (int)std::floor(x) & 255,                  // FIND UNIT CUBE THAT
00050           Y = (int)std::floor(y) & 255,                  // CONTAINS POINT.
00051           Z = (int)std::floor(z) & 255;
00052       x -= std::floor(x);                                // FIND RELATIVE X,Y,Z
00053       y -= std::floor(y);                                // OF POINT IN CUBE.
00054       z -= std::floor(z);
00055       float u = fade(x),                                // COMPUTE FADE CURVES
00056              v = fade(y),                                // FOR EACH OF X,Y,Z.
00057              w = fade(z);
00058       int A = p[X  ]+Y, AA = p[A]+Z, AB = p[A+1]+Z,      // HASH COORDINATES OF
00059           B = p[X+1]+Y, BA = p[B]+Z, BB = p[B+1]+Z;      // THE 8 CUBE CORNERS,
00060 
00061       return lerp(w, lerp(v, lerp(u, grad(p[AA  ], x  , y  , z   ),  // AND ADD
00062                                      grad(p[BA  ], x-1, y  , z   )), // BLENDED
00063                              lerp(u, grad(p[AB  ], x  , y-1, z   ),  // RESULTS
00064                                      grad(p[BB  ], x-1, y-1, z   ))),// FROM  8
00065                      lerp(v, lerp(u, grad(p[AA+1], x  , y  , z-1 ),  // CORNERS
00066                                      grad(p[BA+1], x-1, y  , z-1 )), // OF CUBE
00067                              lerp(u, grad(p[AB+1], x  , y-1, z-1 ),
00068                                      grad(p[BB+1], x-1, y-1, z-1 ))));
00069    }
